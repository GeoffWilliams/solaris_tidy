class solaris_tidy::etc_default_init(
    String $umask = "022",
) {

  # set daemon umask
  shellvar { "/etc/default/init CMASK":
    ensure   => present,
    variable => "CMASK",
    target   => "/etc/default/init",
    value    => $umask,
  }
}
